Листинг программы  clc; clear; close all; f0 = 1800; Vmod = 1200; Vinf = 4800; T = 1 / Vmod; m = Vinf * T; q = 2^m; W = 2 / T; dt = 1/f0/100; t = 0:dt:T; i1 = zeros(q,1); i2 = zeros(q,1); A = 1; s1s2 = zeros(q,2); for c = 1:q i1(c) = floor((c - 1) / sqrt(q)); i2(c) = mod(c - 1,sqrt(q)); s1s2(c,1) = A*(1-((2*i1(c))/(sqrt(q)-1))); s1s2(c, 2) = A*(1-((2*i2(c))/(sqrt(q)-1))); end s = zeros(q,length(t)); for c = 1:q s(c,:) = (s1s2(c,1)*sqrt(W).*cos(2*pi*f0*t)) + (s1s2(c,2)*sqrt(W).*sin(2*pi*f0*t)); end F_1 = sqrt(W)*cos(2*pi*f0*t); F_2 = sqrt(W)*sin(2*pi*f0*t); disp('Проверка ортогональности:') disp(trapz(t, F_1.*F_2)); disp('Проверка ортонормированности:'); disp(trapz(t, F_1.*F_1)); disp(trapz(t, F_2.*F_2)); sij = zeros(q,2); for c = 1:q sij(c,1) = trapz(t,s(c,: ).*F_1); sij(c,2) = trapz(t,s(c,: ).*F_2); end figure(1) plot(sij(:,1),sij(:,2), '.', 'MarkerSize',15); axis([1.5 * min(sij(:,1)), 1.5 * max(sij(:,1)), 1.5 * min(sij(:,2)),1.5 * max(sij(:,2))]); title('Сигнальное созвездие');
